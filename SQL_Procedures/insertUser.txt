create procedure insertUser(@uid int,
							@pass varchar(50),
							@email varchar(50),
							@ltl datetime,
							@fname varchar(50),
							@lname varchar(50),
							@con int,
							@dobb datetime,
							@dojj datetime,
							@status char(1),
							@secQuesId int,
							@secAns varchar(250))
as
INSERT INTO Users VALUES (@uid,@pass,@email,@ltl)
INSERT INTO RegistrationDetails VALUES (@uid,@fname,@lname,@con,@dobb,@dojj,@status,@secQuesId,@secAns)
